def convert_distance_to_float(distance_str):
    try:
        return float(distance_str.replace(',', '.'))
    except ValueError:
        return np.nan

def convert_time_to_minutes(time_str):
    try:
        h, m, s = map(int, time_str.split(':'))
        return h * 60 + m + s / 60
    except ValueError:
        return np.nan

filtered_data_modified['Distance'] = filtered_data_modified['Distance'].apply(convert_distance_to_float)
filtered_data_modified['Time'] = filtered_data_modified['Time'].apply(convert_time_to_minutes)
