sns.get_dataset_names()


tips = sns.load_dataset("tips")

X = tips[["total_bill"]] 
y = tips["tip"]         

lm = LinearRegression()

lm.fit(X, y)

tips['predicted_tip'] = lm.predict(X)


plt.figure(figsize=(10, 6))
sns.scatterplot(data=tips, x='total_bill', y='tip', label='Actual Tips')
sns.lineplot(data=tips, x='total_bill', y='predicted_tip', color='red', label='Regression Line')
plt.title('Simple Linear Regression: Tip vs Total Bill')
plt.legend()
plt.show()
